This project presents an IoT-based patient monitoring and alert system using an ESP8266 NodeMCU and a PIR motion sensor. The system detects patient movement in real time and sends instant alerts to caregivers through the Blynk IoT platform. Although PIR sensors are commonly used in security systems, this project adapts them for biomedical applications such as elderly care, patient activity tracking, and fall detection. The system is low-cost, easy to implement, and suitable for home as well as hospital environments.

Arduino Firmware

Motion_project_final.ino C++ / Arduino 
#define BLYNK_TEMPLATE_ID   "TMPL3qzSUwdO2"
#define BLYNK_TEMPLATE_NAME "Motion detection"
#define BLYNK_AUTH_TOKEN    "x-Zslc4wStJedwIE5CJvXKMVx_SdfDPy"
#define BLYNK_PRINT Serial

#include <ESP8266WiFi.h>
#include <BlynkSimpleEsp8266.h>

char ssid[] = "ABC";        // Your 2.4GHz Wi-Fi name
char pass[] = "12345678";   // Wi-Fi password

int pirPin     = D5;   // PIR motion sensor output pin
int ledPin     = D1;   // LED pin
int motionState = 0;   // Store current motion state

BlynkTimer timer;

// ⚡ Blink LED on motion
void blinkLED() {
  digitalWrite(ledPin, HIGH);
  delay(200);
  digitalWrite(ledPin, LOW);
  delay(200);
}

// ⚡ Check motion sensor state
void checkMotion() {
  int currentState = digitalRead(pirPin);

  if (currentState != motionState) {
    motionState = currentState;

    if (motionState == HIGH) {
      Serial.println("⚠ Motion Detected!");
      Blynk.virtualWrite(V0, 1);
      Blynk.logEvent("motion_detected", "⚠ Motion detected near sensor!");

      // Blink LED 3 times
      for (int i = 0; i < 3; i++) { blinkLED(); }
    } else {
      Blynk.virtualWrite(V0, 0);
      digitalWrite(ledPin, LOW);
    }
  }
}

// ⚙ Auto-reconnect function
void ensureConnection() {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("WiFi disconnected... reconnecting");
    WiFi.begin(ssid, pass);
  }
  if (!Blynk.connected()) {
    Blynk.connect();
  }
}

void setup() {
  Serial.begin(115200);
  pinMode(pirPin, INPUT);
  pinMode(ledPin, OUTPUT);
  Blynk.begin(BLYNK_AUTH_TOKEN, ssid, pass);
  timer.setInterval(500L, checkMotion);
  timer.setInterval(10000L, ensureConnection);
}

void loop() {
  Blynk.run();
  timer.run();
}
